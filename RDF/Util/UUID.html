def self.generate(format: :default)
  begin
    require 'uuid'
    ::UUID.generate(format)
  rescue LoadError => e
    begin
      require 'uuidtools'
      ::UUIDTools::UUID.random_create.hexdigest
    rescue LoadError => e
      raise LoadError.new("no such file to load -- uuid or uuidtools")
    end
  end
end
